maw2_pack.exe dir output.dat
dir - папка с файлами для запаковки БЕЗ слеша на конце
ouput.dat - названия файла куда будет помещен архив

maw2_str_extract.exe [-no] script.scr
-no - если есть, то в базе не будет продублирован извлеченный текст
script.scr - файл скрипта для извлечения текста

maw2_str_insert.exe original.scr new.scr
original.scr - оригинальный файл со скриптом (рядом должен лежать original.xml)
new.scr - новый файл со скриптом
